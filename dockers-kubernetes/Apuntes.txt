// Correr imagen
docker run <imageName> <command>
	// este es un mix entre
	docker run == docker create + docker start
		docker create = crea imagen
		docker start = ejecuta la imagen
	// correr imagen con tag
	docker run elpachanga1/simple-web
	// correr imagen que comparte puertos
	docker run -p 8080:8080 elpachanga1/simple-web

// List images
docker ps	// lista solo imagenes vivas
docker ps --all	// lista historico de movimientos en docker

// Delete all dockers
docker system prune
	// elimina:
	- todos los dockers parados
	- todas las redes sin usar por ninguna imagen
	- todas las imagenes paradas
	- cache builds

// logs
docker logs <ContainerId>

// Stop Container
docker stop <ContainerId>	// solo lo para
docker kill <ContainerId>	// lo para y lo elimina

// Multi-Command container
docker exec -it <ContainerId> <command>

// Build image (Estando en la carpeta donde se encuentra el dockerfile)
docker build .
// Build si quieres que tu imagen tenga un tag para referenciarla despues
	elpachanga1 = tu cuenta docker
	redis-test = tagName
docker build -t elpachanga1/redis-test:latest .

// See all downloaded images
docker images
